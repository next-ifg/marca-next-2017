  \definecolor[named]{descCor}{HTML}{#1}
  \begin{tikzpicture}
    \node (quadro) [%
    rectangle,minimum width=1cm,minimum height=2cm,
    thick,draw=white,fill=descCor,] {};
  \node (texto) [right=6pt of quadro.north east, anchor=north west,] {%
    \begin{tabular}{l l}
      \textbf{HTML} & \##1\\
      \textbf{RGB} & #2\\
      \textbf{CMYK} & #3\\
    \end{tabular}
  };
\end{tikzpicture}
